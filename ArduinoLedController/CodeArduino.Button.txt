//sketch.hex
#define BaudRate 9600
#define LEDPinGreen     10 // pin green LED number
#define LEDPinRed       11 // pin red LED number

char incomingCommand;

void setup()
{
    pinMode(LEDPinGreen, OUTPUT);
    pinMode(LEDPinRed, OUTPUT);
    Serial.begin(BaudRate);
}
void loop()                                 // infinite loop                              
{
    Serial.println("Info from Arduino");    // forward string to usb port

    digitalWrite(LEDPinGreen, HIGH);        // GreenLED ON

     incomingCommand = Serial.read();        //Chech situation
     switch(incomingCommand){
        case '1':
          digitalWrite(LEDPinGreen, LOW);   // GreenLED OFF
          digitalWrite(LEDPinRed, HIGH);    // RedLED ON
          break;
        case '0':
          digitalWrite(LEDPinGreen, LOW);   // GreenLED OFF
          break;
     }
     //delay(1000); //sleep
}
